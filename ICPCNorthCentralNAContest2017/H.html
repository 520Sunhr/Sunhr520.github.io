#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
int n;
cin >> n;
ll ans = 0;
for (int i = n - 1; i >= 0; –i)
{
char ch;
scanf(“ %c”, &ch);
ans += (1LL << i) * (ch == ‘O’);
}
cout << ans << endl;

return 0;

}
