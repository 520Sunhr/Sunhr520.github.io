inline int read()
{
    int ans = 0, mark = 1;
    int t = getchar();
    while (t < '0' || t > '9')
    {
        if (t == '-')
            mark = -1;
        t = getchar();
    }
    while (t <= '9' && t >= '0')
    {
        ans = (ans << 3) + (ans << 1) + t - '0';
        t = getchar();
    }

    return ans * mark;
}

typedef pair<int, int> PII;
const int N = 1e6 + 5;

int n;
PII Te[N];

int main()
{
    in(n);
    for (int i = 0; i < n; i++)
        in(Te[i].first),in(Te[i].second);


    sort(Te, Te + n);

    priority_queue<PII, vector<PII>, greater<PII>> heap;
    for (int i = 0; i < n; i++)
    {
        if (heap.empty() || heap.top().first >= Te[i].first)
        {
            PII stall = { Te[i].second, heap.size() };
            heap.push(stall);
        }
        else
        {
            auto stall = heap.top();
            heap.pop();
            stall.first = Te[i].second;
            heap.push(stall);
        }
    }

    cout << heap.size() << endl;
    return 0;
}
