inline int read()
{
    int ans = 0, mark = 1;
    int t = getchar();
    while(t < '0' || t > '9')
    {
        if (t == '-')
            mark = -1;
        t = getchar();
    }
    while(t <= '9' && t >= '0')
    {
        ans = (ans << 3) + (ans << 1) + t - '0';
        t = getchar();
    }

    return ans * mark;    
}

int qmi(int a,int b,int p)
{
    int res = 1;
    while(b)
    {
        if(b&1) res = (res*a)%p;
        a = a*a%p;
        b >>= 1;
    }
    return res;
}
