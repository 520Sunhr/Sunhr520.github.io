面试题56-I-数组中数字出现的次数

解法一

这个是比较朴素的做法。。但是好像都没有这么做。。而且时间上和空间上都不是很好建议直接看解法二

class Solution {
public:
    vector<int> singleNumbers(vector<int>& nums) {
        //if(nums.size() == 2) return nums;

        bitset<10005> book;// book.reset();
        for(auto s : nums) 
            book.flip(s);

        vector<int> ans;
        for(auto s : nums)
            if(book[s]) ans.push_back(s);

        return ans;
    }
};

解法二

来自万能的评论区,具体原理也可以去万能的评论区找,时间上和空间上都比较好

class Solution {
public:
    vector<int> singleNumbers(vector<int>& nums) {
        int tmp=0;
        for(int num:nums)
            tmp^=num;
        int tmp1=tmp&(-tmp);
        int res1=0,res2=0;
        for(int i:nums)
        {
            if(i&tmp1)
                res1^=i;
            else
                res2^=i;
        }
        return {res1,res2};
    }
};
