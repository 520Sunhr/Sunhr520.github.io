21-合并两个有序链表

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
     ListNode* mergeTwoLists(ListNode* a, ListNode* b){
        if(a == NULL) return b;
        if(b == NULL) return a;

        ListNode* head;ListNode* p;
        if(a->val < b->val) {p = a;a = a->next;}
        else {p = b;b = b->next;}
        p->next = NULL;
        head = p;

        while(a!=NULL&&b!=NULL){
            if(a->val < b->val){
                p->next = a;
                a = a->next;
            }else{
                p->next = b;
                b = b->next;
            }
            p = p->next;
        }

        if(a!=NULL) p->next = a;
        else p-> next = b;

        return head;
    }
};
