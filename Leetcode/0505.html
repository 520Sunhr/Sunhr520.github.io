解法　###

中序遍历

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isValidBST(TreeNode* root) {
        const long long INF = (unsigned long long) -1 >> 1;
        TreeNode* stk[100005]; int tt =  0;
        long long LeftVal = -INF-1;

        while(tt > 0|| root != NULL){
            while(root != NULL){
                stk[++tt] = root;
                root = root->left;
            }
            root = stk[tt--];
            if(root->val <= LeftVal) return false;
            LeftVal = root -> val;
            root = root -> right;
        }
        return true;
    }
};
