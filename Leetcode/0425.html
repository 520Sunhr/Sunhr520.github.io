Leetcode 46 全排列

方法一：直接调用库函数 next_permutation()

class Solution {
public:
    vector<vector<int> > permute(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<vector<int> > ans;
        do{
            ans.push_back(nums);
        }while(next_permutation(nums.begin(), nums.end()));
        return ans;
    }
};
