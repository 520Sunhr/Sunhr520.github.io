983-最低票价

class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        const int N = 365 + 5;
        int dp[N], Map[N]; 
        memset(Map, 0, sizeof Map);
        memset(dp, 0, sizeof dp);
        for(auto d : days) Map[d]=1;

        for(int i = 1; i < N; i++){
            if(!Map[i]) {dp[i] = dp[i -1]; continue;}
            int nums[4];
            nums[1] = dp[i-1]+costs[0];
            nums[2] = i >  7 ? dp[i-7]+costs[1] : costs[1];
            nums[3] = i > 30 ? dp[i-30]+costs[2]: costs[2];
            dp[i] = min(nums[1], min(nums[2], nums[3]));
        }
        return dp[365];
    }
};
