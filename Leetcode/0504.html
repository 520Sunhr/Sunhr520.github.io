45-跳跃游戏II

解法

使用贪心，每次选择最远的

class Solution {
public:
    int jump(vector<int>& nums) {
        int cnt(0), NS(nums.size()), i(0);
        if(NS == 1) return 0;
        if(NS == 2) return 1;
        while(i+1 < NS){
            int j(i+nums[i]), k(j);
            for(int l = i; l <= k&&l<NS; l++)
                if(j<l+nums[l]) i=l,j=l+nums[l];
            if(i<NS) cnt++;
            if(k+1>=NS) break;
        }
        return cnt;
    }
};
