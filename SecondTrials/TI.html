描述

Alice 有一个 N X M 的 01 矩阵
Alice 想知道某些其他的一些 n x m 的 01 矩阵是不是 Alice 的矩阵的子矩阵？

输入

第一行四个整数 N、 M、 n、 m、 分别代表 Alice 的矩阵大小和 Alice 想询问的矩阵大小
接下来 N 行 每行一个长度位 M 的 01 串代表 Alice 的 01 矩阵
第 N+2 行一个整数 q 代表询问次数
每次询问都有 n 行， 每行一个长度为 m 的 01 串代表这次 Alice 询问的矩阵大小

输出

共 q 行， 每行一个“Yes”或“No”代表 Alice 询问的矩阵是或不是 Alice 的矩阵的子矩阵

题目分析

暴力，继续暴力，疯狂暴力！
本身复杂度感觉会很高，但是学长说可以暴力，那就直接暴力！ 
暴力过程：    
    选取每个矩阵的左上角的点进行参考，然后直接遍历所有情况，如果一样，返回true

AC码

#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<vector>
#include<map>
#include <bitset>
using namespace std;

#define INT_MAX 2147483647
#define INT_MIN (- INT_MAX - 1)
#define PI acos(-1)

#define in(x) (x) = read()
#define fre(x) for(int _=(x); _;_--)
#define frein() int in(Loop_var);while(Loop_var--)

inline int read()
{
    int ans = 0, mark = 1;
    int t = getchar();
    while (t < '0' || t > '9')
    {
        if (t == '-')
            mark = -1;
        t = getchar();
    }
    while (t <= '9' && t >= '0')
    {
        ans = (ans << 3) + (ans << 1) + t - '0';
        t = getchar();
    }

    return ans * mark;
}

const int N = 1e3 + 5, M = 105;

bitset<N> Map[N];
bitset<M> AMap[M];
int n, m, x, y;

bool judge(int a, int b)
{
    for (int i = a, k = 1; i <= a + x - 1; i++, k++)
        for (int j = b, t = 1; j <= b + y - 1; j++, t++)
            if (Map[i][j] != AMap[k][t]) return false;

    return true;
}
//检查每个点 

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    //IO

    in(n), in(m), in(x), in(y);

    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= m; j++)
            Map[i][j] = getchar() - '0';
        getchar();
    }

    frein()
    {
        for (int i = 1; i <= x; i++)
        {
            for (int j = 1; j <= y; j++)
                AMap[i][j] = getchar() - '0';
            getchar();
        }

        bool flag = true;
        //用flag记录是否存在 

        for (int i = 1; i <= n - x + 1; i++)
        {
            for (int j = 1; j <= m - y + 1; j++)
                if (judge(i, j))
                {
                    flag = false;
                    break;
                }
            if (!flag) break;
        }//遍历每个左上角的点 

        if (flag) cout << "No" << endl;
        else cout << "Yes" << endl;

    }    

    return 0;
}
